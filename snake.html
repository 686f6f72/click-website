Snake Game Tutorial

Build a classic Snake game using Arduino and a joystick! Guide the snake to eat food, grow longer, and avoid crashing into the walls or itself on a mini OLED screen.

Materials

  • Arduino Uno
  • OLED screen
  • Joystick module
  • 4 jumper wires with double-sided male pins
  • 4 jumper wires with single-sided male pins
  • Breadboard
  • Battery
  • Battery cable
  • USB cable + laptop

Code

#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
// OLED config
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);
// Joystick pins
#define VRX_PIN A0
#define VRY_PIN A1
#define SW_PIN 7
// Game settings
#define SNAKE_SIZE 4
#define MAX_SNAKE_LENGTH 50
int snakeX[MAX_SNAKE_LENGTH], snakeY[MAX_SNAKE_LENGTH];
int snakeLength = 5;
int dirX = 1, dirY = 0;
int foodX, foodY;
int score = 0;
void setup() {
Serial.begin(9600);
// Initialize display
if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
Serial.println(F("SSD1306 allocation failed"));
for (;;);
}
display.clearDisplay();
display.display();
// Joystick as inputs
pinMode(VRX_PIN, INPUT);
pinMode(VRY_PIN, INPUT);
pinMode(SW_PIN, INPUT_PULLUP);
// Initialize snake position
for (int i = 0; i < snakeLength; i++) {
snakeX[i] = SCREEN_WIDTH / 2 - i*SNAKE_SIZE;
snakeY[i] = SCREEN_HEIGHT / 2;
}
placeFood();
}
void loop() {
readJoystick();
moveSnake();
if (checkCollision()) {
gameOver();
} else {
checkFood();
drawGame();
delay(150);
}
}
void readJoystick() {
int x = analogRead(VRX_PIN);
int y = analogRead(VRY_PIN);
if (x < 400 && dirX != 1) { dirX = -1; dirY = 0; }
else if (x > 600 && dirX != -1) { dirX = 1; dirY = 0; }
else if (y < 400 && dirY != 1) { dirX = 0; dirY = -1; }
else if (y > 600 && dirY != -1) { dirX = 0; dirY = 1; }
}
void moveSnake() {
for (int i = snakeLength - 1; i > 0; i--) {
snakeX[i] = snakeX[i - 1];
snakeY[i] = snakeY[i - 1];
}
snakeX[0] += dirX * SNAKE_SIZE;
snakeY[0] += dirY * SNAKE_SIZE;
}
bool checkCollision() {
// Border collision
if (snakeX[0] < 0 || snakeX[0] >= SCREEN_WIDTH ||
snakeY[0] < 0 || snakeY[0] >= SCREEN_HEIGHT) return true;
// Self-collision
for (int i = 1; i < snakeLength; i++) {
if (snakeX[0] == snakeX[i] && snakeY[0] == snakeY[i]) return true;
}
return false;
}
void placeFood() {
foodX = random(0, SCREEN_WIDTH / SNAKE_SIZE) * SNAKE_SIZE;
foodY = random(0, SCREEN_HEIGHT / SNAKE_SIZE) * SNAKE_SIZE;
}
void checkFood() {
if (snakeX[0] == foodX && snakeY[0] == foodY) {
if (snakeLength < MAX_SNAKE_LENGTH) snakeLength++;
score++;
placeFood();
}
}
void drawGame() {
display.clearDisplay();
// Draw food
display.fillRect(foodX, foodY, SNAKE_SIZE, SNAKE_SIZE, SSD1306_WHITE);
// Draw snake
for (int i = 0; i < snakeLength; i++) {
display.fillRect(snakeX[i], snakeY[i], SNAKE_SIZE, SNAKE_SIZE, SSD1306_WHITE);
}
// Display score
display.setTextSize(1);
display.setTextColor(SSD1306_WHITE);
display.setCursor(0,0);
display.print(F("Score: "));
display.print(score);
display.display();
}
void gameOver() {
display.clearDisplay();
display.setTextSize(2);
display.setTextColor(SSD1306_WHITE);
display.setCursor(20, SCREEN_HEIGHT / 2 - 10);
display.println(F("Game Over"));
display.display();
delay(2000);
// Reset
snakeLength = 5;
dirX = 1; dirY = 0;
score = 0;
for (int i = 0; i < snakeLength; i++) {
snakeX[i] = SCREEN_WIDTH / 2 - i*SNAKE_SIZE;
snakeY[i] = SCREEN_HEIGHT / 2;
}
placeFood();
}
